\chapter{
    \textbf{Objectives}
}
\justifying{
    \large{
        \paragraph{} The primary objective of the seminar work is to develop a technique that helps:
        \begin{itemize}
            \item To maintain \textbf{confidentiality} of information
            \item To maintain \textbf{integrity} of information
            \item To maintain \textbf{availability} of information
        \end{itemize}
    }
}
\section{Confidentiality of Information}
\justifying{
    \large{
        \paragraph{} The proposed technique must ensure that information is available only to the authorised users. Hence, it must include
        \begin{itemize}
            \item Protection from unauthorised access and use
            \item Protection of data on systems, in transit and in process
        \end{itemize}
    }
}
\section{Integrity of Information}
\justifying{
    \large{
        \paragraph{} Safeguarding the accuracy and completeness of information is another objective of the proposed technique, and hence it must include,
        \begin{itemize}
            \item Detection of alterations that occurred in storage, transit and process
            \item Prevention of alterations by unauthorised entities
        \end{itemize}
    }
}
\section{Availability of Information}
\justifying{
    \large{
        \paragraph{} The proposed technique must also ensure that the authorised users have access to the information whenever required. Therefore, it must include,
        \begin{itemize}
            \item Fault tolerance
            \item Prevention of data loss and destruction
            \item Acceptable level of performance
        \end{itemize}
    }
}
